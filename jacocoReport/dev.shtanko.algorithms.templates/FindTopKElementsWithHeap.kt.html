package dev.shtanko.algorithms.templates

import java.util.PriorityQueue

private fun fn(arr: IntArray, k: Int): IntArray {
    val criteria = 0 // TODO
    val heap: PriorityQueue<Int> = PriorityQueue(criteria)
    for (num in arr) {
        heap.add(num)
        if (heap.size > k) {
            heap.remove()
        }
    }
    val ans = IntArray(k)
    for (i in 0 until k) {
        ans[i] = heap.remove()
    }
    return ans
}

package dev.shtanko.algorithms.techniques.twopointer

object TwoSums {
    fun twoPointers(arr: IntArray, target: Int): Boolean {
        var pointerOne = 0
        var pointerTwo = arr.size - 1

        while (pointerOne < pointerTwo) {
            val sum = arr[pointerOne] + arr[pointerTwo]
            if (sum == target) {
                return true
            } else if (sum < target) {
                pointerOne++
            } else {
                pointerTwo--
            }
        }
        return false
    }
}
